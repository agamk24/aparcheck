DROP DATABASE IF EXISTS belajar_php;
CREATE DATABASE belajar_php;

USE belajar_php;

CREATE TABLE mahasiswa(
	npm CHAR(12) NOT NULL,
    nama VARCHAR(25),
    alamat TEXT,
    tgl_lahir DATE,
    gender enum('Pria','Wanita'),
    kode_jurusan CHAR(3),
    kota VARCHAR(12),
    PRIMARY KEY(npm)
) ENGINE=InnoDB;

CREATE TABLE admin(
	username VARCHAR(30) NOT NULL,
	password_hash VARCHAR(32),
    PRIMARY KEY (username)
) ENGINE=InnoDB;
